### 8.3 GodLoopPhase — 10-Phase Machine

```typescript
type GodLoopPhase =
  | 'INIT'       // Initialize state, read findings baseline
  | 'AUDIT'      // Run trident-code-audit, compute score
  | 'PLAN'       // Generate build spec from remaining findings
  | 'DISPATCH'   // Dispatch build agents in parallel waves
  | 'VERIFY'     // Verify evidence gate + content hash diff
  | 'COLLECT'    // Synthesize context bridge for next wave
  | 'GAP'        // Analyze gaps, detect stall, problem-solve
  | 'SOLVE'      // Execute revised approach from problem-solving
  | 'LOCK'       // Score >= 96 and gate passed — lock cycle
  | 'FAILED';    // Max cycles reached or unrecoverable error
```

Phase transitions are mechanical and deterministic:

```
INIT -> AUDIT -> PLAN -> DISPATCH -> VERIFY -> COLLECT -> GAP
                                                        |
                                    +-------------------+
                                    |                   |
                               stalled:            improving:
                               SOLVE              AUDIT (next cycle)
                                    |
                               SOLVE -> DISPATCH -> VERIFY -> ...
                                                        |
                                              score >= 96
                                              gate passed
                                                        |
                                                        v
                                                     LOCK
```

### 8.4 PhaseResult Interface

This is the return contract for every `runPhase()` call. The model receives `instructions` as explicit text and acts on them. The orchestrator never executes code directly — it produces instructions that the model executes.

```typescript
interface PhaseResult {
  /** Current phase that was executed */
  phase: GodLoopPhase;
  /** Next phase to execute (the model calls runPhase with this) */
  nextPhase: GodLoopPhase;
  /** Current cycle number (1-indexed) */
  cycle: number;
  /** Current score (0-100), -1 if not yet computed */
  score: number;
  /** Explicit text instructions for the model */
  instructions: string;
  /** Evidence record written to Merkle chain */
  evidence: EvidenceRecord;
  /** Whether state.json was written atomically */
  stateWritten: boolean;
  /** Non-fatal warnings (logged but do not block) */
  warnings: string[];
}
```

### 8.5 runPhase() — Main Entry Point

This is the **only** method that the `trident-poseidon` tool calls. It is the single funnel through which all phases flow.

```typescript
/**
 * Main entry point for Poseidon Mode.
 * Called by trident-poseidon tool with action="start" or action="status".
 *
 * Contract:
 *   1. Read state.json from disk (P1: verify file exists before use)
 *   2. Switch on current phase
 *   3. Execute the mechanical action for that phase
 *   4. Write updated state.json atomically (P5: snapshot-and-rollback)
 *   5. Write evidence to Merkle chain
 *   6. Return PhaseResult with explicit text instructions
 *
 * Error handling (P3): every catch block logs the error,
 *   writes FAILED state if unrecoverable, or recovers with
 *   a degraded instruction set.
 */
async runPhase(targetPath: string, sessionId: string): Promise<PhaseResult> {
  // --- P1: DEFENSIVE READ ---
  const statePath = path.join(targetPath, '.trident', 'god-loop', 'state.json');
  let state: GodLoopState;
  try {
    const raw = fs.readFileSync(statePath, 'utf-8');
    state = JSON.parse(raw) as GodLoopState;
  } catch (err: unknown) {
    // State file missing or corrupt — initialize fresh
    if (err instanceof Error) {
      console.error(`[POSEIDON] Failed to read state: ${err.message}`);
    }
    state = this.initializeState(targetPath, sessionId);
    this.writeStateAtomic(statePath, state);
  }

  // --- P2: PHASE DISPATCH ---
  let result: PhaseResult;
  try {
    switch (state.phase) {
      case 'INIT':
        result = await this.phaseInit(state, targetPath);
        break;
      case 'AUDIT':
        result = await this.phaseAudit(state, targetPath);
        break;
      case 'PLAN':
        result = await this.phasePlan(state, targetPath);
        break;
      case 'DISPATCH':
        result = await this.phaseDispatch(state, targetPath);
        break;
      case 'VERIFY':
        result = await this.phaseVerify(state, targetPath);
        break;
      case 'COLLECT':
        result = await this.phaseCollect(state, targetPath);
        break;
      case 'GAP':
        result = await this.phaseGap(state, targetPath);
        break;
      case 'SOLVE':
        result = await this.phaseSolve(state, targetPath);
        break;
      case 'LOCK':
        result = await this.phaseLock(state, targetPath);
        break;
      case 'FAILED':
        result = await this.phaseFailed(state, targetPath);
        break;
      default:
        // P8: Config validation — unknown phase is unrecoverable
        state.phase = 'FAILED';
        state.error = `Unknown phase: ${state.phase}`;
        this.writeStateAtomic(statePath, state);
        result = this.buildResult(
          state, 'FAILED',
          `UNRECOVERABLE: Unknown phase. State marked as FAILED.`,
          [],
        );
    }
  } catch (err: unknown) {
    // P3: ERROR COMPLETENESS — log, write FAILED state, return error result
    const msg = err instanceof Error ? err.message : String(err);
    console.error(`[POSEIDON] Phase ${state.phase} failed: ${msg}`);
    state.phase = 'FAILED';
    state.error = msg;
    state.failedAt = new Date().toISOString();
    this.writeStateAtomic(statePath, state);
    result = this.buildResult(
      state, 'FAILED',
      `PHASE ${state.phase.toUpperCase()} CRASHED: ${msg}. State written as FAILED. Check logs.`,
      [`Phase crash: ${msg}`],
    );
  }

  // --- P4: STATE + EVIDENCE WRITES ---
  // State was already written inside the phase method.
  // Write evidence to Merkle chain.
  try {
    await this.evidenceStore.append({
      phase: state.phase,
      cycle: state.cycle,
      score: result.score,
      timestamp: new Date().toISOString(),
      action: `Phase ${result.phase} to ${result.nextPhase}`,
    });
  } catch (err: unknown) {
    const msg = err instanceof Error ? err.message : String(err);
    result.warnings.push(`Evidence write failed: ${msg}`);
    console.warn(`[POSEIDON] Evidence write failed: ${msg}`);
  }

  return result;
}
```

### 8.6 Enhanced extractScore() — Hash-Based Progressive Scoring

The existing `extractScore()` reads score from audit output text via regex. This is **preserved** as a sanity check. A new `computeScore()` method (Section 9) produces the **authoritative** hash-based score.

```typescript
/**
 * ENHANCED extractScore — now delegates to computeScore() for the
 * authoritative score, while keeping the regex-based score as a
 * sanity check. If they diverge by more than 20 points, a warning
 * is logged.
 *
 * OLD behavior: read "Score: X/100" from audit output text
 * NEW behavior: compute hash-based score, cross-check with regex
 *
 * @param auditOutput  Raw text from trident-code-audit
 * @param preAudit     Finding signatures from baseline audit
 * @param postAudit    Finding signatures from current audit
 * @returns            { score: number, divergence: number | null }
 */
extractScore(
  auditOutput: string,
  preAudit: FindingSignature[],
  postAudit: FindingSignature[],
): { score: number; divergence: number | null } {
  // --- Regex score (sanity check) ---
  // P2: Type certainty — parse with fallback
  let regexScore = -1;
  const match = auditOutput.match(/Score:\s*(\d+)\s*\/\s*100/);
  if (match && match[1]) {
    const parsed = parseInt(match[1], 10);
    if (!Number.isNaN(parsed) && parsed >= 0 && parsed <= 100) {
      regexScore = parsed;
    }
  }

  // --- Hash-based score (authoritative) ---
  const hashResult = this.computeScore(preAudit, postAudit);
  const authoritativeScore = hashResult.score;

  // --- Divergence check ---
  let divergence: number | null = null;
  if (regexScore >= 0) {
    divergence = Math.abs(authoritativeScore - regexScore);
    if (divergence > 20) {
      console.warn(
        `[POSEIDON] SCORE DIVERGENCE: hash=${authoritativeScore}, regex=${regexScore}, ` +
        `delta=${divergence}. Using hash-based score as authoritative.`,
      );
    }
  }

  return { score: authoritativeScore, divergence };
}
```

### 8.7 Enhanced verifyCycle() — Evidence Gate + Content Hash Diff

The existing `verifyCycle()` checks if the cycle should advance. It is **enhanced** to call `evidenceStore.verifyAndGate()` (Bug Fix #3 — this method existed but was never called).

```typescript
/**
 * ENHANCED verifyCycle — now performs three checks:
 *   1. Merkle chain integrity (calls verifyAndGate — bug fix #3)
 *   2. Content hash diff on all modified files
 *   3. passRate >= 0.96 threshold
 *
 * OLD behavior: checked passRate from a field, no chain verification
 * NEW behavior: full gate verification with tamper detection
 *
 * @param state   Current GodLoopState
 * @returns       { advance: boolean, reason: string, details: EvidenceGateDetails }
 */
async verifyCycle(
  state: GodLoopState,
): Promise<{ advance: boolean; reason: string; details: EvidenceGateDetails | null }> {

  // --- CHECK 1: Merkle chain integrity ---
  const gateResult = await this.evidenceStore.verifyAndGate();

  if (!gateResult.passed) {
    // Evidence chain is broken or incomplete.
    // Severity determines recovery action.
    if (gateResult.passRate < 0.50) {
      // Severe corruption — mark FAILED, cannot trust any evidence
      state.phase = 'FAILED';
      state.error = `Severe evidence corruption: passRate=${gateResult.passRate}`;
      this.writeStateAtomic(state.statePath, state);
      return {
        advance: false,
        reason: `SEVERE CORRUPTION: Evidence gate failed at passRate=${gateResult.passRate}. ` +
                `Broken at entry ${gateResult.details.firstBrokenEntry}. Cycle FAILED.`,
        details: gateResult.details,
      };
    }
    // Partial corruption (0.50-0.95) — re-dispatch the wave
    return {
      advance: false,
      reason: `EVIDENCE GATE FAILED. Re-dispatching wave ${state.wave}. ` +
              `Pass rate: ${gateResult.passRate}. Broken at entry ${gateResult.details.firstBrokenEntry}.`,
      details: gateResult.details,
    };
  }

  // --- CHECK 2: Content hash diff on modified files ---
  const modifiedFiles = state.modifiedFiles ?? [];
  let hashMismatches = 0;

  for (const filePath of modifiedFiles) {
    // P1: Defensive — verify file exists
    if (!fs.existsSync(filePath)) {
      console.warn(`[POSEIDON] Modified file no longer exists: ${filePath}`);
      hashMismatches++;
      continue;
    }

    const currentHash = this.hashFile(filePath);
    const recordedHash = state.fileHashes.get(filePath);

    if (recordedHash && currentHash !== recordedHash) {
      // File changed since dispatch — expected (builder modified it)
      state.fileHashes.set(filePath, currentHash);
    } else if (!recordedHash) {
      // New file — record its hash
      state.fileHashes.set(filePath, currentHash);
    }
  }

  if (hashMismatches > modifiedFiles.length * 0.5) {
    return {
      advance: false,
      reason: `More than 50% of modified files are missing. Re-dispatching wave ${state.wave}.`,
      details: gateResult.details,
    };
  }

  // --- CHECK 3: passRate threshold ---
  // passRate comes from the score engine (closed + modified * 0.5 / total)
  if (state.passRate < 0.96) {
    return {
      advance: false,
      reason: `Pass rate ${state.passRate.toFixed(4)} < 0.96 threshold. ` +
              `Score: ${state.score}/100. Continuing to next cycle.`,
      details: gateResult.details,
    };
  }

  // All checks passed — cycle can advance or lock
  return {
    advance: true,
    reason: `All gates passed. passRate=${state.passRate.toFixed(4)}, ` +
            `chain intact, ${modifiedFiles.length} files verified.`,
    details: gateResult.details,
  };
}
```

### 8.8 detectStall() — Score Stagnation Detection

Stall detection prevents infinite loops where the builder makes changes that do not improve the score. A stall is defined as no score improvement for 3 consecutive cycles.

```typescript
/**
 * Detect score stagnation using CycleTracker history.
 *
 * Stall = no score improvement for STALL_THRESHOLD (3) consecutive cycles.
 * "Improvement" means the score increased by at least STALL_MIN_GAIN (1 point).
 * A score that stays flat or decreases counts as no improvement.
 *
 * @returns  StallAnalysis with cyclesSinceImprovement and diagnostic text
 */
detectStall(): StallAnalysis {
  const STALL_THRESHOLD = 3;
  const STALL_MIN_GAIN = 1; // points

  const history = this.cycleTracker.getHistory();

  if (history.length < STALL_THRESHOLD + 1) {
    return {
      stalled: false,
      cyclesSinceImprovement: history.length,
      analysis: `Insufficient history (${history.length} cycles). ` +
                `Need ${STALL_THRESHOLD + 1} to detect stall.`,
    };
  }

  let cyclesSinceImprovement = 0;
  const recent = history.slice(-(STALL_THRESHOLD + 1));

  // Walk backward from most recent, counting cycles without improvement
  for (let i = recent.length - 1; i > 0; i--) {
    const current = recent[i].score;
    const previous = recent[i - 1].score;
    const gain = current - previous;

    if (gain >= STALL_MIN_GAIN) {
      break; // Found improvement — stop counting
    }
    cyclesSinceImprovement++;
  }

  const stalled = cyclesSinceImprovement >= STALL_THRESHOLD;

  const analysis = stalled
    ? `STALLED: Score has not improved by >= ${STALL_MIN_GAIN} point for ` +
      `${cyclesSinceImprovement} consecutive cycles. ` +
      `Score history: [${recent.map(c => c.score).join(', ')}]. ` +
      `Problem-solving is REQUIRED before continuing.`
    : `Progressing: ${cyclesSinceImprovement} cycles since last improvement ` +
      `(threshold: ${STALL_THRESHOLD}). Score history: ` +
      `[${recent.map(c => c.score).join(', ')}].`;

  return { stalled, cyclesSinceImprovement, analysis };
}
```

The `StallAnalysis` interface:

```typescript
interface StallAnalysis {
  /** True if no improvement for >= 3 cycles */
  stalled: boolean;
  /** How many cycles since the last score improvement */
  cyclesSinceImprovement: number;
  /** Human-readable diagnostic with score history */
  analysis: string;
}
```

### 8.9 runProblemSolve() — Invoke 6-Layer Mental Framework

When `detectStall()` returns `stalled: true`, the GAP phase calls `runProblemSolve()`. This invokes `trident-problem-solving` with a pre-built mental framework — it **never guesses**.

```typescript
/**
 * Invoke trident-problem-solving when score has stalled.
 *
 * Constructs a pre-built mental framework from:
 *   - Cycle history (what scores were achieved)
 *   - Persistent findings (what refuses to be fixed)
 *   - Failed approaches (what the builder already tried)
 *
 * The 6-layer pipeline (assumption -> action -> observe -> gap -> meta -> verify)
 * produces a revised working plan that the SOLVE phase executes.
 *
 * @param state     Current GodLoopState
 * @param stall     StallAnalysis from detectStall()
 * @returns         Revised approach with specific action items
 */
async runProblemSolve(
  state: GodLoopState,
  stall: StallAnalysis,
): Promise<ProblemSolveResult> {

  // Gather persistent findings (appeared in all recent cycles with same hash)
  const persistentFindings = this.findingTracker.getPersistentFindings();

  // Gather failed approaches (findings that were dispatched but not resolved)
  const failedApproaches = this.findingTracker.getFailedApproaches();

  // Build the problem description
  const problem = `Score has stalled at ${state.score}/100 for ` +
    `${stall.cyclesSinceImprovement} cycles. ` +
    `${persistentFindings.length} findings are persistent (unchanged hash across cycles). ` +
    `${failedApproaches.length} approaches have been tried without resolution.`;

  // Build the reasoning chain (evidence-based, no guessing)
  const reasoning: string[] = [
    `Cycle history: ${this.cycleTracker.getHistory().map(c => `C${c.cycle}=${c.score}`).join(', ')}`,
    `Persistent findings: ${persistentFindings.slice(0, 10).map(f => `${f.id} (${f.issue})`).join(', ')}`,
    `Failed approaches: ${failedApproaches.slice(0, 5).map(a => `${a.findingId}: ${a.approach}`).join(', ')}`,
    `Pattern analysis: ${persistentFindings.filter(f => f.issue === 'THEATRICAL_CODE').length} ` +
      `theatrical, ${persistentFindings.filter(f => f.severity === 'critical').length} critical`,
    `Architecture constraints: ${state.architectureConstraints?.join('; ') ?? 'none identified'}`,
  ];

  // Build the working plan template (problem-solving will refine this)
  const workingPlan: string[] = [
    `Re-audit files containing persistent findings with trident-code-audit for fresh signatures`,
    `Group persistent findings by root cause (not by file)`,
    `For each root cause group, design a single fix that addresses ALL findings in the group`,
    `If a finding has been persistent for 5+ cycles, consider it architectural — ` +
      `document as known-issue and exclude from score`,
    `Re-dispatch with revised build spec targeting root causes, not symptoms`,
  ];

  // Call trident-problem-solving tool (via task subagent)
  const solveResult = await this.invokeProblemSolving({
    targetPath: state.targetPath,
    problem,
    reasoning,
    workingPlan,
  });

  return {
    revisedApproach: solveResult.workingPlan,
    rootCauseAnalysis: solveResult.rootCauseAnalysis,
    keyInsights: solveResult.insights,
    newFindings: solveResult.newFindings ?? [],
    shouldExclude: solveResult.shouldExclude ?? [],
  };
}
```

The return type:

```typescript
interface ProblemSolveResult {
  /** Revised working plan from the 6-layer pipeline */
  revisedApproach: string[];
  /** Root cause analysis (the "why" behind the stall) */
  rootCauseAnalysis: string;
  /** Key insights discovered during problem-solving */
  keyInsights: string[];
  /** New findings discovered during analysis */
  newFindings: FindingSignature[];
  /** Findings that should be excluded (architectural, 5+ cycles persistent) */
  shouldExclude: string[];
}
```

```typescript
/**
```

#### 8.10.2 phaseAudit()

```typescript
/**
 * AUDIT phase: Run full audit, compute progressive score.
 *
 * Calls: trident-code-audit (full mode) + computeScore()
 * Writes: state.json with updated findings, score, passRate
 * Returns: instructions to proceed to PLAN or LOCK
 */
private async phaseAudit(state: GodLoopState, targetPath: string): Promise<PhaseResult> {
  // 2. Call trident-code-audit full mode
  const fullAudit = await this.runCodeAudit(targetPath, 'full');

  // 3. Parameters: targetPath, mode='full'
  // 4. Process: extract finding signatures for current cycle
  const currentFindings = this.extractFindingSignatures(fullAudit, state.cycle);

  // Compute hash-based score
  const scoreResult = this.computeScore(
    state.baselineFindings,
    currentFindings,
  );

  // Cross-check with regex (sanity)
  const { score, divergence } = this.extractScore(
    fullAudit.outputText,
    state.baselineFindings,
    currentFindings,
  );

  // 5. State writes
  state.score = score;
  state.currentFindings = currentFindings;
  state.passRate = scoreResult.closed / Math.max(scoreResult.total, 1);
  state.lastAuditAt = new Date().toISOString();

  // Update finding tracker
  this.findingTracker.update(currentFindings, state.cycle);

  // Record score in cycle tracker
  this.cycleTracker.record(state.cycle, score);

  // Determine next phase
  const warnings: string[] = [];
  if (divergence !== null && divergence > 20) {
    warnings.push(`Score divergence: hash=${score}, regex delta=${divergence}`);
  }

  let nextPhase: GodLoopPhase;
  let instructions: string;

  if (score >= 96 && state.passRate >= 0.96) {
    // Score threshold met — advance to VERIFY for gate check
    state.phase = 'VERIFY';
    nextPhase = 'VERIFY';
    instructions =
      `## AUDIT COMPLETE — THRESHOLD MET\n\n` +
      `Score: ${score}/100\n` +
      `Closed: ${scoreResult.closed}, Modified: ${scoreResult.modified}, ` +
      `Persistent: ${scoreResult.persistent}, Total: ${scoreResult.total}\n` +
      `Pass rate: ${state.passRate.toFixed(4)}\n\n` +
      `## NEXT ACTION\n` +
      `Call trident-poseidon action="start" targetPath="${targetPath}"\n` +
      `Phase advances to VERIFY for evidence gate check.`;
  } else {
    // Continue to PLAN
    state.phase = 'PLAN';
    nextPhase = 'PLAN';
    instructions =
      `## AUDIT COMPLETE\n\n` +
      `Score: ${score}/100\n` +
      `Closed: ${scoreResult.closed}, Modified: ${scoreResult.modified}, ` +
      `Persistent: ${scoreResult.persistent}, Total: ${scoreResult.total}\n` +
      `Pass rate: ${state.passRate.toFixed(4)}\n\n` +
      `## NEXT ACTION\n` +
      `Call trident-poseidon action="start" targetPath="${targetPath}"\n` +
      `Phase advances to PLAN.`;
  }

  this.writeStateAtomic(state.statePath, state);
  return this.buildResult(state, nextPhase, instructions, warnings);
}
```

#### 8.10.3 phasePlan()

```typescript
/**
 * PLAN phase: Generate build spec from remaining findings.
 *
 * Calls: trident-deep-planning with remaining findings + architecture
 * Writes: state.json with wave manifest path
 * Returns: instructions to proceed to DISPATCH
 */
private async phasePlan(state: GodLoopState, targetPath: string): Promise<PhaseResult> {
  // 2. Call trident-deep-planning
  const remainingFindings = state.currentFindings.filter(
    f => f.status === 'pending' || f.status === 'persistent',
  );

  const planResult = await this.invokeDeepPlanning({
    targetPath,
    requirements: `Fix ${remainingFindings.length} findings. ` +
      `Current score: ${state.score}/100. Target: >= 96.`,
    findings: remainingFindings,
    architecture: state.architectureConstraints ?? [],
  });

  // 3. Parameters: targetPath, findings, architecture constraints
  // 4. Process: parse plan into wave manifest
  const waveManifest = this.generateWaveManifest(
    planResult.buildSpec,
    remainingFindings,
    state.cycle,
    state.wave,
    targetPath,
  );

  // Write wave manifest to disk
  const manifestPath = path.join(
    targetPath, '.trident', 'god-loop',
    `wave-manifest-c${state.cycle}-w${state.wave}.json`,
  );
  this.writeStateAtomic(manifestPath, waveManifest);

  // 5. State writes
  state.phase = 'DISPATCH';
  state.waveManifestPath = manifestPath;
  state.wave += 1;
  this.writeStateAtomic(state.statePath, state);

  // 6. Text instruction
  const instructions =
    `## PLAN COMPLETE\n\n` +
```

## SECTION 9: SCORE ENGINE

### 9.1 Overview

The Score Engine replaces the broken "count remaining findings" approach with a **hash-based progressive scoring** system. The old approach was fundamentally flawed: with 868 findings (many in bundle noise), fixing 100 left the score at 0 because the denominator never changed. The new approach tracks **which specific findings were actually addressed** by comparing source code hashes before and after each build wave.

### 9.2 FindingSignature Interface

Every finding tracked by the Score Engine is represented as a `FindingSignature`. This is the atomic unit of scoring.

```typescript
interface FindingSignature {
  /** Unique finding ID (generated from file + line + issue type) */
  id: string;
  /** Source file path (relative to target) */
  file: string;
  /** Line number where the finding was detected */
  line: number;
  /** Issue type (e.g., 'EMPTY_CATCH', 'THEATRICAL_CODE', 'MISSING_RETURN') */
  issue: string;
  /** Severity: 'critical' | 'high' | 'medium' | 'low' */
  severity: string;
  /** Confidence score 0.0-1.0 from the audit engine */
  confidence: number;
  /**
   * SHA-256 hash of surrounding code block (+/- 5 lines from the finding).
   * This detects whether the builder ACTUALLY modified the code.
   * If hash is unchanged across cycles -> 'persistent'.
   * If hash changed -> 'modified'.
   * If finding no longer appears -> 'closed'.
   */
  sourceHash: string;
  /** File type classification for weighting */
  fileType: 'source' | 'bundle';
  /** Lifecycle status */
  status: 'pending' | 'modified' | 'closed' | 'persistent';
  /** First cycle this finding was observed */
  firstSeenAt: number;
  /** Most recent cycle this finding was observed */
  lastSeenAt: number;
  /** Whether a fix was dispatched to a build agent */
  fixAttempted: boolean;
  /** Whether the fix was verified by post-audit */
  fixVerified: boolean;
}
```

### 9.3 computeScore() Method

```typescript
/**
 * Compute hash-based progressive score.
 *
 * For each pre-audit finding, checks if it still exists in post-audit:
 *   - sourceHash changed -> 'modified' (builder touched it)
 *   - finding gone -> 'closed' (resolved)
 *   - same sourceHash -> 'persistent' (unfixed)
 *
 * Progressive score = (closed + modified * 0.5) / totalSourceFindings * 100
 *
 * @param preAudit   Finding signatures from baseline audit
 * @param postAudit  Finding signatures from current audit
 * @returns          Score breakdown with closed/modified/persistent counts
 */
function computeScore(
  preAudit: FindingSignature[],
  postAudit: FindingSignature[],
): { score: number; closed: number; modified: number; persistent: number; total: number } {

  // Index post-audit findings by ID for O(1) lookup
  const postMap = new Map<string, FindingSignature>();
  for (const finding of postAudit) {
    postMap.set(finding.id, finding);
  }

  let closed = 0;      // Finding resolved (not in post-audit)
  let modified = 0;    // Finding touched (hash changed)
  let persistent = 0;  // Finding unfixed (same hash)
  let totalSourceFindings = 0;

  for (const preFinding of preAudit) {
    // --- Confidence floor: exclude low-confidence findings ---
    if (preFinding.confidence < 0.85) {
      continue; // Not counted in score
    }

    // --- Type weighting: source findings only in denominator ---
    if (preFinding.fileType === 'bundle') {
      continue; // Bundle findings do not count in denominator
    }
    // fileType === 'source' -> counts
    totalSourceFindings++;

    const postFinding = postMap.get(preFinding.id);

    if (!postFinding) {
      // Finding no longer exists — resolved
      closed++;
      preFinding.status = 'closed';
      preFinding.fixVerified = true;
    } else if (postFinding.sourceHash !== preFinding.sourceHash) {
      // Finding still exists but code changed — builder modified it
      modified++;
      preFinding.status = 'modified';
      preFinding.sourceHash = postFinding.sourceHash;
      preFinding.lastSeenAt = postFinding.lastSeenAt;
    } else {
      // Finding exists with identical hash — builder did not touch it
      persistent++;
      preFinding.status = 'persistent';
      preFinding.lastSeenAt = postFinding.lastSeenAt;
    }
  }

  // --- Progressive score ---
  // (closed findings count fully, modified count half)
  const denominator = Math.max(totalSourceFindings, 1); // P2: prevent div-by-zero
  const rawScore = ((closed + modified * 0.5) / denominator) * 100;

  // Round to nearest integer
  const score = Math.round(rawScore);

  return { score, closed, modified, persistent, total: totalSourceFindings };
}
```

### 9.4 Hash Function for Finding Signatures

```typescript
import * as crypto from 'crypto';
import * as fs from 'fs';
import * as path from 'path';

/**
 * Hash the code context surrounding a finding.
 *
 * Takes the file path, line number, and +/- 5 lines of surrounding code,
 * concatenates them, and computes SHA-256.
 *
 * This hash is used to detect whether the builder ACTUALLY modified
 * the code near a finding:
 *   - Unchanged hash -> finding is 'persistent' (builder did not touch it)
 *   - Changed hash -> finding is 'modified' (builder made a change)
 *
 * @param file         Relative file path
 * @param line         Line number of the finding
 * @param codeContext  The +/- 5 lines of surrounding code (pre-extracted)
 * @returns            Hex-encoded SHA-256 hash
 */
function hashFindingContext(file: string, line: number, codeContext: string): string {
  // Concatenate: file path + line number + code context
  // This ensures that the same issue in different files or at different
  // lines produces different hashes.
  const input = `${file}:${line}:${codeContext}`;

  // SHA-256 hash
  return crypto.createHash('sha256').update(input, 'utf-8').digest('hex');
}

/**
 * Extract +/- 5 lines of code context from a file for hashing.
 *
 * @param filePath  Absolute path to the source file
 * @param line      Line number of the finding (1-indexed)
 * @returns         The code context string, or empty string on error
 */
function extractCodeContext(filePath: string, line: number): string {
  const CONTEXT_RADIUS = 5;

  try {
    // P1: Defensive — verify file exists
    if (!fs.existsSync(filePath)) {
      return '';
    }

    const content = fs.readFileSync(filePath, 'utf-8');
    const lines = content.split('\n');

    // P2: Type certainty — validate line number
    const startLine = Math.max(0, line - 1 - CONTEXT_RADIUS);
    const endLine = Math.min(lines.length, line + CONTEXT_RADIUS);

    return lines.slice(startLine, endLine).join('\n');
  } catch (err: unknown) {
    // P3: Error completeness — log and return empty
    const msg = err instanceof Error ? err.message : String(err);
    console.warn(`[SCORE_ENGINE] Failed to extract context from ${filePath}: ${msg}`);
    return '';
  }
}
```

### 9.5 Progressive Scoring: Old vs New

**OLD approach (broken):**

```
Score = 100 - (remainingFindings / totalFindings * 100)
```

Problems:
- With 868 total findings (many bundle noise), fixing 100 source findings
  leaves: `100 - (768 / 868 * 100) = 100 - 88 = 12/100`
- The score barely moves because bundle noise inflates the denominator.
- Builders see no progress -> motivation drops -> loop stalls.
- No distinction between "builder touched it" and "builder fixed it."

**NEW approach (progressive):**

```
Score = (closed + modified * 0.5) / totalSourceFindings * 100
```

Where:
- `closed` = findings that no longer appear in post-audit (resolved)
- `modified` = findings whose sourceHash changed (builder touched them)
- `totalSourceFindings` = high-confidence findings in `.ts` files only
- Bundle (`.js`) findings are excluded from the denominator

Results:
- Fixed 100 of 200 source findings -> score = 50
- Fixed 150 of 200 -> score = 75
- Fixed 190 of 200 -> score = 95
- Fixed 193 of 200 -> score = 96.5 ~ 97 -> LOCKED

The score **progresses** with each fix and does not stagnate due to bundle noise.

### 9.6 Type Weighting Table

| File Type | Extension | Weight | Rationale |
|---|---|---|---|
| Source | `.ts` | 1.0 | Real source code, findings matter, builder can fix |
| Bundle | `.js` (minified) | 0.1 | Generated code, findings are noise, builder cannot fix |
| Declaration | `.d.ts` | 0.0 | Type declarations only, not executable |
| Config | `.json` | 0.0 | Not executable code, no runtime impact |
| Map | `.js.map` | 0.0 | Source maps, debugging artifacts only |

The type is determined by file extension at signature creation time:

```typescript
function classifyFileType(filePath: string): 'source' | 'bundle' {
  if (filePath.endsWith('.ts') && !filePath.endsWith('.d.ts')) {
    return 'source';
  }
  if (filePath.endsWith('.d.ts')) {
    return 'bundle'; // Treated as 0.0 weight, excluded from score
  }
  if (filePath.endsWith('.js') || filePath.endsWith('.mjs')) {
    // Check if minified (single-line, >500 chars per line average)
    return 'bundle';
  }
  return 'bundle'; // Default: anything else is non-source
}
```

### 9.7 Confidence Floor Specification

Not all findings are equal. Low-confidence findings from the audit engine are often false positives. The confidence floor prevents them from polluting the score:

| Confidence Range | Action | Score Impact |
|---|---|---|
| `< 0.50` | **EXCLUDED** entirely | Not counted in any calculation |
| `0.50 - 0.84` | **LOGGED** but not dispatched | Appears in audit report, does not affect score |
| `>= 0.85` | **DISPATCHED** to build agents | Counts in score denominator and numerator |

The audit engine still **reports** all findings (for completeness), but only high-confidence findings drive the God Loop. This prevents the loop from chasing phantom issues that cannot be verified.

```typescript
/**
 * Apply confidence floor to findings.
 * Returns only findings that meet the dispatch threshold.
 *
 * @param findings  All findings from the audit
 * @returns         High-confidence findings (>= 0.85)
 */
function applyConfidenceFloor(findings: FindingSignature[]): FindingSignature[] {
  return findings.filter(f => {
    if (f.confidence < 0.50) {
      // Excluded entirely — not even logged for tracking
      return false;
    }
    if (f.confidence < 0.85) {
      // Logged but not dispatched
      console.debug(
        `[SCORE_ENGINE] Low-confidence finding logged but excluded: ` +
        `${f.id} (${f.issue}, confidence=${f.confidence.toFixed(2)})`,
      );
      return false;
    }
    return true; // confidence >= 0.85 — dispatched
  });
}
```

### 9.8 FindingTracker — Lifecycle Management

The `FindingTracker` class manages the lifecycle of findings across cycles:

```typescript
class FindingTracker {
  private findings: Map<string, FindingSignature> = new Map();
  private excludedIds: Set<string> = new Set();
  private failedApproaches: Map<string, Array<{ cycle: number; approach: string }>> = new Map();

  /**
   * Update findings from the latest audit.
   * Tracks which findings appeared, changed, or disappeared.
   */
  update(currentFindings: FindingSignature[], cycle: number): void {
    const currentIds = new Set(currentFindings.map(f => f.id));

    // Update existing findings
    for (const finding of currentFindings) {
      const existing = this.findings.get(finding.id);
      if (existing) {
        existing.lastSeenAt = cycle;
        if (finding.sourceHash !== existing.sourceHash) {
          existing.sourceHash = finding.sourceHash;
          existing.status = 'modified';
        } else if (existing.status !== 'closed') {
          // Same hash, still present — persistent
          existing.status = 'persistent';
        }
      } else {
        // New finding
        finding.firstSeenAt = cycle;
        finding.lastSeenAt = cycle;
        finding.status = 'pending';
        this.findings.set(finding.id, finding);
      }
    }

    // Findings that disappeared -> closed
    for (const [id, finding] of this.findings) {
      if (!currentIds.has(id) && finding.status !== 'closed') {
        finding.status = 'closed';
        finding.fixVerified = true;
      }
    }
  }

  /**
   * Get findings that have been persistent (same hash) for >= threshold cycles.
   */
  getPersistentFindings(minCycles: number = 3): FindingSignature[] {
    return Array.from(this.findings.values()).filter(f => {
      if (this.excludedIds.has(f.id)) return false;
      if (f.status !== 'persistent') return false;
      return (f.lastSeenAt - f.firstSeenAt) >= minCycles;
    });
  }

  /**
   * Get approaches that were tried but did not resolve the finding.
   */
  getFailedApproaches(): Array<{ findingId: string; approach: string }> {
    const result: Array<{ findingId: string; approach: string }> = [];
    for (const [findingId, attempts] of this.failedApproaches) {
      for (const attempt of attempts) {
        result.push({ findingId, approach: attempt.approach });
      }
    }
    return result;
  }

  /**
   * Mark a finding as excluded (architectural, cannot be fixed).
   * Excluded findings are removed from the score denominator.
   */
  excludeFinding(findingId: string): void {
    this.excludedIds.add(findingId);
    const finding = this.findings.get(findingId);
    if (finding) {
      finding.status = 'closed'; // Treated as resolved for scoring
      finding.fixVerified = true;
    }
  }

  /**
   * Record that a fix approach was attempted for a finding.
   */
  recordFixAttempt(findingId: string, cycle: number, approach: string): void {
    if (!this.failedApproaches.has(findingId)) {
      this.failedApproaches.set(findingId, []);
    }
    this.failedApproaches.get(findingId)!.push({ cycle, approach });
  }
}
```

### 10.3 verifyAndGate() — New Method on EvidenceStore

```typescript
/**
 * Verify the Merkle chain and apply the evidence gate.
 *
 * Walks all entries, verifies hash links and recomputation.
 * If passRate < 0.96, the gate fails.
 *
 * This method is called by GodLoopOrchestrator.verifyCycle().
 *
 * @returns  Gate result with pass/fail decision and detailed diagnostics
 */
async verifyAndGate(): Promise<{
  passed: boolean;
  passRate: number;
  brokenAt: number | null;
  details: EvidenceGateDetails;
}> {
  // Read all nodes from disk
  const nodes = this.readAllNodes();

  if (nodes.length === 0) {
    // Empty chain — trivially valid (first entry has not been written yet)
    return {
      passed: true,
      passRate: 1.0,
      brokenAt: null,
      details: {
        totalEntries: 0,
        validEntries: 0,
        invalidEntries: 0,
        firstBrokenEntry: null,
        chainIntact: true,
        hashMismatches: [],
      },
    };
  }

  let validCount = 0;
  const hashMismatches: Array<{ entryId: number; expected: string; actual: string }> = [];
  let firstBrokenEntry: number | null = null;

  for (let i = 0; i < nodes.length; i++) {
    const node = nodes[i];
    let isValid = true;

    // --- CHECK 1: Previous hash links correctly ---
    if (i === 0) {
      // Genesis node — prevHash must be all zeros
      if (node.prevHash !== '0'.repeat(64)) {
        isValid = false;
        hashMismatches.push({
          entryId: i,
          expected: '0'.repeat(64),
          actual: node.prevHash,
        });
      }
    } else {
      // Non-genesis — prevHash must match previous node hash
      const prevNode = nodes[i - 1];
      if (node.prevHash !== prevNode.hash) {
        isValid = false;
        hashMismatches.push({
          entryId: i,
          expected: prevNode.hash,
          actual: node.prevHash,
        });
      }
    }

    // --- CHECK 2: Current hash matches recomputation ---
    const recomputedHash = crypto
      .createHash('sha256')
      .update(node.prevHash + JSON.stringify(node.data), 'utf-8')
      .digest('hex');

    if (node.hash !== recomputedHash) {
      isValid = false;
      hashMismatches.push({
        entryId: i,
        expected: recomputedHash,
        actual: node.hash,
      });
    }

    if (isValid) {
      validCount++;
    } else if (firstBrokenEntry === null) {
      firstBrokenEntry = i;
    }
  }

  const totalEntries = nodes.length;
  const invalidEntries = totalEntries - validCount;
  const passRate = totalEntries > 0 ? validCount / totalEntries : 1.0;
  const passed = passRate >= 0.96;
  const chainIntact = invalidEntries === 0;

  // --- Tamper-proof audit trail ---
  // If gate failed, write evidence of the failure to the chain itself.
  // This creates an immutable record that the gate was checked and failed.
  if (!passed) {
    try {
      await this.append({
        phase: 'VERIFY' as GodLoopPhase,
        cycle: -1, // Indicates gate check, not a cycle
        score: -1,
        timestamp: new Date().toISOString(),
        action: `EVIDENCE GATE FAILED: passRate=${passRate.toFixed(4)}, ` +
          `brokenAt=${firstBrokenEntry}, invalid=${invalidEntries}`,
      });
    } catch (err: unknown) {
      // P3: Cannot write evidence of failure — log but continue
      const msg = err instanceof Error ? err.message : String(err);
      console.error(`[EVIDENCE_STORE] Failed to write gate-failure evidence: ${msg}`);
    }
  }

  return {
    passed,
    passRate,
    brokenAt: firstBrokenEntry,
    details: {
      totalEntries,
      validEntries: validCount,
      invalidEntries,
      firstBrokenEntry,
      chainIntact,
      hashMismatches,
    },
  };
}
```

## SECTION 11: CONTEXT BRIDGE WIRING

### 11.1 Overview

The existing `context-synthesis-engine.ts` implements a 4-layer pipeline (Collect -> Score -> Compress -> Inject). In v4.4.1, this engine is **wired into** the COLLECT phase of the God Loop. This bridges context between waves, ensuring that each wave agents receive the relevant findings, architecture changes, and failed approaches from previous waves.

### 11.2 Context Bridge Architecture

```
WAVE N completes (all agents returned)
  |
  v
COLLECT phase runs
  |
  +-- context-synthesis-engine.collect()
  |     Gathers:
  |       - Wave N build results (which files changed)
  |       - Wave N audit findings (what is still broken)
  |       - Wave N evidence records (Merkle entries)
  |       - Previous wave context (T1 from wave N-1)
  |
  +-- context-synthesis-engine.score()
  |     Ranks by relevance:
  |       - Unresolved critical findings: HIGH priority
  |       - Newly introduced findings: HIGH priority
  |       - Resolved findings: LOW priority (context only)
  |       - Architecture changes: MEDIUM priority
  |
  +-- context-synthesis-engine.compress()
  |     Fits to token budget:
  |       - T1 injectable: < 5K chars (operational instructions)
  |       - T2 knowledge file: < 40K chars (architectural understanding)
  |
  +-- context-synthesis-engine.inject()
        Writes to disk:
          - T1 -> .trident/god-loop/context/wave-N-T1.md
          - T2 -> .trident/god-loop/context/wave-N-T2.md (milestones only)
  |
  v
Wave N+1 agents receive T1 in their dispatch prompt
Wave N+1 orchestrator reads T2 for architectural context
```

### 11.3 generateContextBridge() Method

```typescript
/**
 * Generate T1 and T2 context bridges between waves.
 *
 * Uses context-synthesis-engine to:
 *   1. Collect all relevant context from the current wave
 *   2. Score by relevance to the next wave
 *   3. Compress to T1 (< 5K chars) and T2 (< 40K chars)
 *   4. Write to disk for injection
 *
 * @param state       Current GodLoopState
 * @param targetPath  Project root path
 * @returns           Paths to written T1 and T2 files
 */
async generateContextBridge(
  state: GodLoopState,
  targetPath: string,
): Promise<{
  t1Path: string;
  t2Path: string | null;
  warnings: string[];
}> {
  const warnings: string[] = [];
  const contextDir = path.join(targetPath, '.trident', 'god-loop', 'context');

  // Ensure context directory exists
  // P1: Defensive — verify parent exists
  if (!fs.existsSync(path.dirname(contextDir))) {
    fs.mkdirSync(path.dirname(contextDir), { recursive: true });
  }
  if (!fs.existsSync(contextDir)) {
    fs.mkdirSync(contextDir, { recursive: true });
  }

  // --- 1. COLLECT ---
  const collected = await this.contextEngine.collect({
    cycle: state.cycle,
    wave: state.wave,
    currentFindings: state.currentFindings ?? [],
    baselineFindings: state.baselineFindings ?? [],
    modifiedFiles: state.modifiedFiles ?? [],
    score: state.score,
    previousT1Path: state.t1ContextPath,
    architectureConstraints: state.architectureConstraints ?? [],
    revisedApproach: state.revisedApproach ?? [],
  });

  // --- 2. SCORE ---
  const scored = await this.contextEngine.score(collected);

  // --- 3. COMPRESS ---
  // Determine T1/T2 ratio based on cycle number
  const ratio = this.getContextRatio(state.cycle);

  const t1Budget = Math.floor(5000 * ratio.t1Percent);
  const t2Budget = Math.floor(40000 * ratio.t2Percent);

  const t1Content = await this.contextEngine.compress(scored, {
    mode: 'T1',
    maxChars: t1Budget,
    focus: 'operational',
  });

  // T2 only on milestones
  const t2Milestone = this.shouldGenerateT2(state);
  let t2Content: string | null = null;
  if (t2Milestone) {
    t2Content = await this.contextEngine.compress(scored, {
      mode: 'T2',
      maxChars: t2Budget,
      focus: 'architectural',
    });
  }

  // --- 4. INJECT (write to disk) ---
  const t1Path = path.join(contextDir, `wave-${state.wave}-T1.md`);
  fs.writeFileSync(t1Path, this.formatT1(state, t1Content), 'utf-8');

  let t2Path: string | null = null;
  if (t2Content) {
    t2Path = path.join(contextDir, `wave-${state.wave}-T2.md`);
    fs.writeFileSync(t2Path, this.formatT2(state, t2Content), 'utf-8');
  }

  return { t1Path, t2Path, warnings };
}
```

### 11.5 T2 Generation Triggers

T2 knowledge files are only generated when a milestone is reached. This prevents token budget waste on routine cycles.

| Trigger | T2 Content | Condition |
|---|---|---|
| Score improvement > 10 points | Architecture progress summary, what worked, what did not | `score - prevScore > 10` |
| Cycle completion (full loop) | Full cycle report, all findings resolved/persistent, approach analysis | Phase returns to AUDIT (cycle +1) |
| Wave completion (all agents returned) | Wave results, build quality assessment, recommendations | VERIFY phase passes |
| New critical finding introduced | Regression analysis, root cause, containment strategy | New critical finding in post-audit |
| Problem-solving output | Revised approach, key insights, decision chain | SOLVE phase completes |

```typescript
/**
 * Determine if a T2 milestone has been reached.
 * T2 is only generated when significant architectural
 * understanding changes — not on every cycle.
 */
private shouldGenerateT2(state: GodLoopState): boolean {
  const history = this.cycleTracker.getHistory();

  // Trigger 1: Score improvement > 10 points
  if (history.length >= 2) {
    const current = history[history.length - 1].score;
    const previous = history[history.length - 2].score;
    if (current - previous > 10) return true;
  }

  // Trigger 2: Cycle completion (full loop)
  if (state.phase === 'GAP' || state.phase === 'AUDIT') {
    return true;
  }

  // Trigger 3: New critical finding introduced
  const newCritical = state.currentFindings?.filter(
    f => f.severity === 'critical' && f.firstSeenAt === state.cycle,
  );
  if (newCritical && newCritical.length > 0) return true;

  // Trigger 4: Problem-solving output
  if (state.revisedApproach && state.revisedApproach.length > 0) {
    return true;
  }

  return false;
}
```

### 11.6 T1/T2 Ratio Specification

|---|---|---|---|
| 1-3 (early) | 70% | 30% | More operational: agents need immediate instructions, not architecture |
| 4-10 (mid) | 60% | 40% | Balanced: agents need both operations and architecture |
| 11+ (late) | 50% | 50% | More architectural: deep problem-solving, root cause analysis |

### 11.7 Compaction Recovery Context Injection

When the conversation is compacted (context window fills up), the `compactingHook` fires. This hook reconstructs the Poseidon state from disk and injects a recovery message into the conversation — **not** the system prompt.

```typescript
 * Compaction recovery for Poseidon Mode.
 *
 * On compaction event, this hook fires and:
 *   1. Reads current state.json from disk
 *   2. Reads latest T1 from .trident/god-loop/context/
 *   3. Reads latest T2 if a milestone was reached
 *   4. Constructs a recovery message
 *   5. Injects into messages (NOT system prompt)
 *
 * This ensures the God Loop survives compaction without losing state.

function poseidonCompactionRecovery(targetPath: string): string | null {
  const statePath = path.join(targetPath, '.trident', 'god-loop', 'state.json');

  // P1: Defensive — verify state file exists
  if (!fs.existsSync(statePath)) {
    return null; // Not in Poseidon Mode

  // P2: Type certainty — parse with validation
  let state: GodLoopState;
  try {
    state = JSON.parse(fs.readFileSync(statePath, 'utf-8'));

    return null; // Corrupt state — cannot recover
  }

  // Read T1 context
  let t1Content = '';
  if (state.t1ContextPath && fs.existsSync(state.t1ContextPath)) {
    t1Content = fs.readFileSync(state.t1ContextPath, 'utf-8');
  }

  // Read T2 context if available
  let t2Content = '';
  if (state.t2ContextPath && fs.existsSync(state.t2ContextPath)) {
    t2Content = fs.readFileSync(state.t2ContextPath, 'utf-8');
  }

  // Construct recovery message
  const recoveryMessage = `POSEIDON MODE ACTIVE. You are in Cycle ${state.cycle}, Phase ${state.phase}.
Score: ${state.score}/100. Wave: ${state.wave}.

${t1Content}

${t2Content ? '## Architectural Context\n' + t2Content + '\n' : ''}

Read .trident/god-loop/state.json for full state. Continue the GOD loop.
Do NOT stop until LOCKED or FAILED.

## NEXT ACTION
Call trident-poseidon action="start" targetPath="${targetPath}"`;

  return recoveryMessage;
}
```

### 12.3 Wave Manifest Generation (PLAN Phase)

The wave manifest is generated in five steps during the PLAN phase:

```typescript
/**
 * Generate a wave manifest from build spec and remaining findings.
 *
 * Steps:
 *   1. trident-deep-planning generates build spec from remaining findings
 *   2. Build spec parsed into individual agent tasks
 *   3. Tasks grouped by file (same file -> same agent)
 *   4. Dependencies resolved (if B imports A, B depends on A)
 *   5. Independent tasks assigned to same wave
 *
 * @param buildSpec          Output from trident-deep-planning
 * @param remainingFindings  Findings not yet resolved
 * @param cycle              Current cycle number
 * @param wave               Current wave number
 * @param targetPath         Project root
 * @returns                  WaveManifest for dispatch
 */
function generateWaveManifest(
  buildSpec: BuildSpec,
  remainingFindings: FindingSignature[],
  cycle: number,
  wave: number,
  targetPath: string,
): WaveManifest {
  // --- STEP 1: Already done (buildSpec passed in from deep-planning) ---

  // --- STEP 2: Parse build spec into individual tasks ---
  const tasks = parseBuildSpecToTasks(buildSpec, remainingFindings);

  // --- STEP 3: Group tasks by file ---
  const fileGroups = new Map<string, AgentTask[]>();
  for (const task of tasks) {
    const group = fileGroups.get(task.filePath) ?? [];
    group.push(task);
    fileGroups.set(task.filePath, group);
  }

  // --- STEP 4: Resolve dependencies ---
  const dependencies = resolveDependencies(fileGroups, targetPath);

  // --- STEP 5: Assign to wave (all independent tasks -> this wave) ---
  const agents: WaveAgent[] = [];
  let agentId = 1;

  for (const [filePath, taskGroup] of fileGroups) {
    // Skip agents whose dependencies are in this same wave
    // (they will be dispatched in a later wave)
    const agentDeps = dependencies.get(`agent-${String(agentId).padStart(3, '0')}`) ?? [];
    const hasSameWaveDep = agentDeps.some(depId => {
      // Check if depId maps to an agent already in this wave
      return fileGroups.has(getFilePathForAgent(depId, agents));
    });

    if (hasSameWaveDep) {
      continue; // Defer to next wave
    }

    const allFindings = taskGroup.flatMap(t => t.findings);
    const spec = taskGroup.map(t => t.description).join('\n');

    agents.push({
      id: `agent-${String(agentId).padStart(3, '0')}`,
      module: extractModuleName(filePath),
      filePath,
      spec,
      findings: allFindings,
      allowedTools: determineAllowedTools(allFindings),
      bashAccess: needsBashAccess(allFindings),
    });

    agentId++;
  }

  // --- Determine T1 context path ---
  const t1ContextPath = path.join(
    targetPath, '.trident', 'god-loop', 'context',
    `wave-${wave}-T1.md`,
  );

  return {
    cycle,
    wave,
    targetPath,
    agents,
    dependencies,
    t1ContextPath,
    createdAt: Date.now(),
  };
}
```

### 12.4 Dependency Resolution

Dependencies are resolved by analyzing import graphs. If agent B file imports from agent A file, B depends on A.

```typescript
/**
 * Resolve dependencies between agents based on import graph.
 *
 * If agent B file imports from agent A file, B depends on A.
 * This prevents agents from modifying files that depend on
 * not-yet-modified interfaces.
 *
 * @param fileGroups  Map of filePath -> tasks
 * @param targetPath  Project root
 * @returns           Dependency graph: agentId -> [dependency agentIds]
 */
function resolveDependencies(
  fileGroups: Map<string, AgentTask[]>,
  targetPath: string,
): Map<string, string[]> {
  const dependencies = new Map<string, string[]>();
  const filePaths = Array.from(fileGroups.keys());

  // Build a reverse index: filePath -> agentId
  const fileToAgent = new Map<string, string>();
  let id = 1;
  for (const filePath of filePaths) {
    fileToAgent.set(filePath, `agent-${String(id).padStart(3, '0')}`);
    id++;
  }

  // For each file, check its imports
  for (const filePath of filePaths) {
    const agentId = fileToAgent.get(filePath)!;
    const deps: string[] = [];

    const imports = extractImports(path.join(targetPath, filePath));
    for (const importPath of imports) {
      const resolvedImport = resolveImportPath(importPath, filePath, targetPath);
      const depAgentId = fileToAgent.get(resolvedImport);
      if (depAgentId && depAgentId !== agentId) {
        deps.push(depAgentId);
      }
    }

    dependencies.set(agentId, deps);
  }

  return dependencies;
}
```

### 12.5 Wave Manifest Usage in DISPATCH Phase

The wave manifest is consumed in the DISPATCH phase. The orchestrator reads it and generates explicit text instructions for the model. The model then dispatches agents by calling `task()` for each one.

```typescript
/**
 * Build dispatch instructions from a wave manifest.
 *
 * The MODEL reads these instructions and dispatches agents
 * by calling task() for each agent in ONE message.
 * The orchestrator does NOT dispatch agents itself.
 */
function buildDispatchInstructions(
  manifest: WaveManifest,
  targetPath: string,
): string {
  const agentBlocks = manifest.agents.map(agent => {
    const findingSummary = agent.findings.length > 0
      ? agent.findings.map(f => `  - ${f.id}: ${f.issue} (${f.severity}) at ${f.file}:${f.line}`).join('\n')
      : '  (none — architectural task)';

    return `### Agent ${agent.id}: ${agent.module}
- **File:** ${agent.filePath}
- **Task:** ${agent.spec}
- **Findings to address (${agent.findings.length}):**
${findingSummary}
- **Allowed tools:** ${agent.allowedTools.join(', ')}
- **Bash access:** ${agent.bashAccess ? 'YES' : 'NO'}
- **Context bridge:** Read T1 at ${manifest.t1ContextPath}`;
  }).join('\n\n');

  return `## WAVE ${manifest.wave} DISPATCH

Dispatch ${manifest.agents.length} agents in parallel by calling task() ${manifest.agents.length} times in ONE message.

${agentBlocks}

## AFTER ALL AGENTS RETURN
Call trident-poseidon action="start" targetPath="${targetPath}"
Phase will advance to VERIFY for evidence gate check.`;
}
```

### 12.6 Allowed Tools Determination

```typescript
/**
 * Determine which tools an agent is allowed to use based on
 * the findings it is addressing.
 *
 * @param findings  Findings assigned to this agent
 * @returns         List of allowed tool names
 */
function determineAllowedTools(findings: FindingSignature[]): string[] {
  const tools = ['read', 'grep', 'glob']; // Always allowed

  // Agents that fix source code need edit access
  const hasSourceFindings = findings.some(f => f.fileType === 'source');
  if (hasSourceFindings) {
    tools.push('edit', 'write');
  }

  // Agents with complexity findings may need to restructure
  const hasComplexityIssues = findings.some(
    f => f.issue.includes('COMPLEXITY') || f.issue.includes('LENGTH'),
  );
  if (hasComplexityIssues) {
    tools.push('write'); // Full rewrite access for restructuring
  }

  return tools;
}

/**
 * Determine if an agent needs bash access.
 * Bash access is granted for agents that need to compile or test.
 *
 * @param findings  Findings assigned to this agent
 * @returns         True if bash access is needed
 */
function needsBashAccess(findings: FindingSignature[]): boolean {
  // Agents fixing build-chain issues need bash to verify compilation
  return findings.some(f =>
    f.issue.includes('BUILD') ||
    f.issue.includes('COMPILE') ||
    f.issue.includes('IMPORT') ||
    f.issue.includes('DEPENDENCY'),
  );
}
```

### 12.8 Wave Manifest Validation

Before a wave manifest is used in DISPATCH, it is validated to ensure correctness:

```typescript
/**
 * Validate a wave manifest before dispatch.
 *
 * Checks:
 *   1. Every agent has at least one finding or a spec
 *   2. No two agents modify the same file
 *   3. All file paths are within targetPath (no escape)
 *   4. All dependency agent IDs exist in the manifest
 *
 * @param manifest    Wave manifest to validate
 * @returns           { valid: boolean, errors: string[] }
 */
function validateWaveManifest(manifest: WaveManifest): {
  valid: boolean;
  errors: string[];
} {
  const errors: string[] = [];

  // Check 1: Every agent has findings or spec
  for (const agent of manifest.agents) {
    if (agent.findings.length === 0 && (!agent.spec || agent.spec.trim().length === 0)) {
      errors.push(`Agent ${agent.id} has no findings and no spec`);
    }
  }

  // Check 2: No two agents modify the same file
  const filePaths = new Set<string>();
  for (const agent of manifest.agents) {
    if (filePaths.has(agent.filePath)) {
      errors.push(`Multiple agents modify ${agent.filePath}`);
    }
    filePaths.add(agent.filePath);
  }

  // Check 3: All file paths within targetPath
  for (const agent of manifest.agents) {
    const resolved = path.resolve(manifest.targetPath, agent.filePath);
    if (!resolved.startsWith(path.resolve(manifest.targetPath))) {
      errors.push(`Agent ${agent.id} file path escapes target: ${agent.filePath}`);
    }
  }

  // Check 4: All dependency IDs exist
  const agentIds = new Set(manifest.agents.map(a => a.id));
  for (const [agentId, deps] of manifest.dependencies) {
    if (!agentIds.has(agentId)) {
      errors.push(`Dependency references unknown agent: ${agentId}`);
    }
    for (const depId of deps) {
      if (!agentIds.has(depId)) {
        errors.push(`Agent ${agentId} depends on unknown agent: ${depId}`);
      }
    }
  }

  return { valid: errors.length === 0, errors };
}
```
